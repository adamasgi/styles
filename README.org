I namespaced them with 'ctâ€™ to hopefully avoid any collisions.

	/**/

	--ct-main-bg: ;

	--ct-alt-bg: ;

	/**/

	/**/
	--ct-error: ;
	--ct-warning: ;
	--ct-info: ;
	/**/
	--ct-important: ;
	--ct-unimportant: ;
}
#+END_SRC
